Test 2 - Add Minecount in Class Box - Solution

class Box:

    def __init__(self, boxType, row, column):
        self.boxType = boxType
        self.row = row
        self.column = column
        self.minecount = 0
        
        startRow = row - 1
        stopRow = row + 2
        startColumn = column - 1
        stopColumn = column + 2
        if startRow < 0:
            startRow = 0
        if stopRow > 10:
            stopRow = 10
        if startColumn < 0:
            startColumn = 0
        if stopColumn > 10:
            stopColumn = 10
        for i in range(startRow, stopRow):
            for j in range(startColumn, stopColumn):
                if gameMapStrings[i][j] == "M":
                    self.minecount = self.minecount + 1
