        if words[i] == "one":

def getCartInfo():
  if not cart:
    return "Cart is empty"
  
  products = cart.keys()
  info = "Cart: "
  totalCost = 0
  for p in products:
    info = info + str(cart[p]) + " " + p + "s "
    totalCost = totalCost + cart[p] * prices[p]
