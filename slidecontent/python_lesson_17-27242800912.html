class Box:
    boxType = "E"
    row = 0
    column = 0

    def __init__(self, boxType, row, column):
        self.boxType = boxType
        self.row = row
        self.column = column

# TODO: complete method Draw
#       this method when called should use function canvas.drawImage
#       to draw the image of the Box based on boxType
#       and the x & y positions of top left corner, and the width & height value
#       should be passed into this method and used in canvas.drawImage
    def Draw(self, x, y, w, h):
        if self.boxType == "E":
            canvas.drawImage("E0.png", x, y, w, h)
        else:
            canvas.drawImage("M.png", x, y, w, h)

A = Box("M", 0, 0)

# TODO: You need to pass the x & y positions of top left corner (0, 0),
