async function getKickShotCommandHorizontal(ball_id, pocket_id, strength) {
   const targetBallPos = Balls[ball_id];
   const contactPos = extrapolatePoints(Pockets[pocket_id], Balls[ball_id], BallDiameter);
   // TODO: set mirrorLine to LEFT_X by default, and change it to RIGHT_X if pocket_id is 0 or 5
   let mirrorLine = TOP_Y;
   if ([0, 1, 2].includes(pocket_id)) mirrorLine = BOTTOM_Y;

   // TODO: update calculation of mirrorPos, so its y coordinate is the same as contactPos,
   //       and its x coordinate is 2 times mirrorLine then subtracts contactPos.x
   const mirrorPos = {
      x: contactPos.x,  y: 2 * mirrorLine - contactPos.y
   };

   // compose shot command object using mirrorPos
   const cmd = { aimx: mirrorPos.x, aimy: mirrorPos.y, strength: strength, targetBallID: ball_id, targetPocketID: pocket_id };

   for (let k=0; k<5; k++) {
     // calculate closest trajectory point
     const closestPoint = await calculateCTP(cmd, 0, contactPos);

     // TODO: calculate errorInY using y coordinates
     const errorInX = Math.abs(closestPoint.x - contactPos.x);

     // TODO: change all references to x to y, and use errorInY as well
     if (closestPoint.x > contactPos.x) {
      cmd.aimx = cmd.aimx - errorInX;
     } else {
      cmd.aimx = cmd.aimx + errorInX; 
     }
   }
   cmd.probability = await calculateProbability(cmd);
   return cmd;
}
